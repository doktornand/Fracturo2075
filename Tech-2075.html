(defun starts-with? (str prefix)
   (and (>= (length str) (length prefix))
        (string= str prefix :end2 (length prefix))))





(defun extract-between (str start-char end-char &optional (start-pos 0))
   (let ((s-pos (position start-char str :start start-pos))
         (e-pos (position end-char str :start (1+ start-pos))))
     (when (and s-pos e-pos)
       (subseq str (1+ s-pos) e-pos))))





(defun parse-header (line)
   (cond
     ((starts-with? line "; FICHIER")
      (:timestamp ,(read-from-string (subseq line (search "STEIN " line) nil :start1 0))))
     ((starts-with? line "; Cœur")
      (let* ((coeur-str (extract-between line #\: #\|))
             (delta-str (subseq line (+ (search "|" line) 2))))
        (:coeur ,(read-from-string (string-trim " " coeur-str))
          :delta ,(read-from-string delta-str))))
     (t nil)))





(defun classify-line (line)
   (cond
     ((search "ANCRAGE" line) :anchor)
     ((search "•••" line) :silence)
     ((search "Ω<" line) :fracturo-invocation)
     ((search "defmacro" line) :lisp-macro)
     (t :unknown)))





(defun parse-entry (line)
   (let* ((type (classify-line line))
          (ts-str (extract-between line #\[ #\]))
          (timestamp (when ts-str (read-from-string ts-str)))
          (content line)
          (lisp-recon nil)
          (warns nil))





    ;; Transduction Fracturo → Lisp
     (when (eq type :fracturo-invocation)
       (let* ((phase (extract-between line #\< #\>))
              (v-pos (search "v" line :start2 (search ">" line)))
              (version (when v-pos (read-from-string (subseq line v-pos (+ v-pos 2)))))
              (lieu-pos (search "—" line))
              (lieu (when lieu-pos (string-trim " " (subseq line (+ lieu-pos 1) (search "—" line :start2 lieu-pos)))))
              (param (subseq line (+ (or lieu-pos 0) 2))))
         (setf lisp-recon
               (,(cdr (assoc phase fracturo->lisp-map :test #'string-equal))
                 ,(intern (string-upcase lieu) :keyword)
                 :version ,version :param ,(string-trim " •" param)))
         (push "Invocation Fracturo détectée" warns)))





    (make-fzo-entry :timestamp timestamp
                     :raw-line line
                     :type type
                     :content content
                     :lisp-reconstruction lisp-recon
                     :warnings warns)))





(defun parse-fzo (filepath)
   (let ((entries nil)
         (metadata nil))
     (with-open-file (stream filepath :direction :input)
       (loop for line = (read-line stream nil nil)
             while line do
             (let ((parsed (parse-header line)))
               (when parsed
                 (setf metadata (merge 'list metadata parsed :test #'equal))
                 (loop-finish)))))
       ;; Relire pour les entrées (deuxième passe ou reset si possible, ici on simplifie en relisant)
       (file-position stream 0)
       (loop for line = (read-line stream nil nil)
             while line do
             (unless (starts-with? line ";")
               (push (parse-entry line) entries))))
     (make-session-audit
      :timestamp (getf metadata :timestamp)
      :coeur-declare (getf metadata :coeur)
      :delta-final (getf metadata :delta)
      :entries (nreverse entries)
      :violations nil
      :metrics nil
      :status :PENDING)))

(defun verify-session (audit)
   (let* ((entries (session-audit-entries audit))
          (violations nil)
          (max-delta 0.0)
          (open-contexts 0)
          (skeith-count 0)
          (corbenik-count 0))





    (dolist (entry entries)
       (let ((type (fzo-entry-type entry)))
         ;; 1. Vérification des contextes ouverts/fermés
         (when (eq type :fracturo-invocation) (incf open-contexts))
         (when (eq type :silence) (decf open-contexts))





        ;; 2. Comptage des macros critiques
         (when (search "SKEITH" (fzo-entry-raw-line entry)) (incf skeith-count))
         (when (search "CORBENIK" (fzo-entry-raw-line entry)) (incf corbenik-count))





        ;; 3. Détection de Δ implicite via sauts de timestamp
         (when (and (eq type :silence) (> open-contexts 0))
           (push "⚠️ Contexte non clôturé avant •••" violations))





        ;; 4. Validation Cœur
         (when (search "COEUR-NOIR" (fzo-entry-raw-line entry))
           (push "🔴 INTENTION COEUR NOIR DÉTECTÉE" violations))))





    ;; Règles métier
     (when (> open-contexts 0)
       (push "⚠️ ~D invocation(s) sans clôture •••" violations))
     (when (> skeith-count (+ corbenik-count 2))
       (push "🔴 Déséquilibre skeith/corbenik (destruction sans recompilation)" violations))





    (setf (session-audit-violations audit) (nreverse violations)
           (session-audit-metrics audit) (:open-contexts ,open-contexts
                                             :skeith-count ,skeith-count
                                             :corbenik-count ,corbenik-count)
           (session-audit-status audit)
           (cond ((null violations) :VALID)
                 ((some #'starts-with? violations "🔴") :COMPROMISED)
                 (t :WARNING)))
     audit))
